### 2.2 Plugin 入口 (index.ts)

```typescript
import type { OpenClawPluginAPI } from "openclaw";
import { registerSceneTools }      from "./tools/scene";
import { registerGameObjectTools } from "./tools/gameobject";
import { registerAssetTools }      from "./tools/asset";
import { registerEditorTools }     from "./tools/editor";
import { registerBuildTools }      from "./tools/build";
import { createUnityClient }       from "./unity-client";

export default function register(api: OpenClawPluginAPI) {
  const config = api.getConfig();
  const client = createUnityClient({
    port:    config.port    ?? 23456,
    timeout: config.timeout ?? 10000,
  });

  registerSceneTools(api, client);
  registerGameObjectTools(api, client);
  registerAssetTools(api, client);
  registerEditorTools(api, client);
  registerBuildTools(api, client);
}
```

#### MainThreadDispatcher.cs

```csharp
public static class MainThreadDispatcher {
    // 线程安全队列
    private static readonly ConcurrentQueue<(Func<object> action, TaskCompletionSource<object> tcs)> _queue = new();

    // 在 EditorApplication.update 中调用
    public static void Tick() {
        while (_queue.TryDequeue(out var item)) {
            try   { item.tcs.SetResult(item.action()); }
            catch (Exception ex) { item.tcs.SetException(ex); }
        }
    }

    // 后台线程调用，阻塞等待主线程执行
    public static T Dispatch<T>(Func<T> action) {
        var tcs = new TaskCompletionSource<object>(TaskCreationOptions.RunContinuationsAsynchronously);
        _queue.Enqueue((() => action(), tcs));
        return (T)tcs.Task.GetAwaiter().GetResult();
    }
}
```

### 3.3 Handler 设计模式

所有 Handler 遵循统一模式：

```csharp
public class GameObjectHandler {
    public object HandleCreate(HttpContext ctx) {
        // 1. 解析请求参数
        var req = ctx.ParseBody<CreateGameObjectRequest>();

        // 2. 参数验证
        if (string.IsNullOrEmpty(req.Name))
            return ApiResponse.Error("INVALID_PARAMS", "name is required");

        // 3. 调度到主线程执行 Unity API
        return MainThreadDispatcher.Dispatch(() => {
            // 4. 执行 Unity 操作（主线程）
            var go = new GameObject(req.Name);
            if (!string.IsNullOrEmpty(req.ParentPath)) {
                var parent = GameObject.Find(req.ParentPath);
                if (parent == null)
                    throw new UnityApiException("OBJECT_NOT_FOUND", $"Parent '{req.ParentPath}' not found");
                go.transform.SetParent(parent.transform);
            }
            Undo.RegisterCreatedObjectUndo(go, $"Create {req.Name}");

            // 5. 返回结果
            return ApiResponse.Ok(new { id = go.GetInstanceID(), path = GetObjectPath(go) });
        });
    }
}
```

## 6. 扩展性设计

### 6.1 新增 Unity 操作

1. 在 Unity C# 侧 `Handlers/` 添加新方法或新 Handler 类
2. 在 `RequestRouter` 注册新路由
3. 在 TypeScript Plugin 的 `tools/` 中添加对应 `api.registerTool()`
4. 更新 `skills/unity-editor/references/tools-reference.md`

### 6.2 支持多 Unity 实例

当前设计单实例，扩展多实例需要：
- 端口发现机制（扫描端口范围或读取注册表文件）
- 工具参数中增加 `instanceId` 选择目标

### 6.3 WebSocket 事件推送（核心功能，随 HTTP 同期实现）

WebSocket 与 HTTP REST **同期开发**，是代码自修正循环的必要支撑，不再是可选扩展。

#### 职责划分

```
HTTP REST  →  Agent 主动发起的操作（写脚本、查场景、创建对象等）
WebSocket  →  Unity 主动推送事件（编译完成/失败、Console 日志、Play Mode 变化）
```

#### 版本兼容实现

```csharp
// 统一接口，条件编译隔离两套实现，对上层 EventBroadcaster 透明
public interface IWebSocketServer {
    void Start(int port);   // 接收 HTTP 端口，内部自动使用 port+1 作为 WS 端口
    void Broadcast(string json);
    void Stop();
}

#if UNITY_2022_3_OR_NEWER
// 使用 .NET 内置实现
public class BuiltinWebSocketServer : IWebSocketServer { ... }
#else
// 引入 websocket-sharp（MIT, ~200KB）
public class SharpWebSocketServer : IWebSocketServer { ... }
#endif
```

Unity Editor Server 启动时根据版本自动选择实现，外部行为完全一致。

#### OpenClaw Plugin 侧 WebSocket 管理

```typescript
class UnityWebSocketClient {
  private ws: WebSocket | null = null;

  connect(wsPort: number) {
    this.ws = new WebSocket(`ws://127.0.0.1:${wsPort}/ws`);
    this.ws.on("message", (raw) => {
      const { event, data } = JSON.parse(raw.toString());
      this.emit(event, data);
    });
    // 断连后 2s 自动重连（处理 Domain Reload 窗口期）
    this.ws.on("close", () => setTimeout(() => this.connect(wsPort), 2000));
  }

  waitForEvent(eventName: string, timeoutMs = 60000): Promise<unknown> {
    return new Promise((resolve, reject) => {
      const timer = setTimeout(
        () => reject(new Error(`Timeout waiting for ${eventName}`)), timeoutMs
      );
      this.once(eventName, (data) => { clearTimeout(timer); resolve(data); });
    });
  }
}
```
